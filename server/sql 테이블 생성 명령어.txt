CREATE TABLE userdata (
  NO INT PRIMARY KEY AUTO_INCREMENT,
  ID TEXT,
  PASSWORD TEXT,
  NAME TEXT,
  AGE INTEGER,
  EMAIL TEXT,
  DEVICE TEXT
) ENGINE=INNODB;

CREATE TABLE token (
  `id` int(10) NOT NULL AUTO_INCREMENT,
  Token varchar(300) NOT NULL,
  userId TEXT,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB  DEFAULT CHARSET=latin1 AUTO_INCREMENT=1;
